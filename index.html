function doGet(e) { try { const sheetId = e.parameter.sheetId; if (!sheetId) { return ContentService.createTextOutput(JSON.stringify({ ok: true, msg: 'API activa' })) .setMimeType(ContentService.MimeType.JSON); } const sheet = SpreadsheetApp.openById(sheetId).getActiveSheet(); const valores = sheet.getDataRange().getValues(); if (valores.length <= 1) { return ContentService.createTextOutput(JSON.stringify({ ok: true, data: [] })) .setMimeType(ContentService.MimeType.JSON); } // Fecha de hoy en formato DD/MM/YYYY const ahora = new Date(); const diaHoy = String(ahora.getDate()).padStart(2, '0'); const mesHoy = String(ahora.getMonth() + 1).padStart(2, '0'); const anioHoy = ahora.getFullYear(); const hoyStr = diaHoy + '/' + mesHoy + '/' + anioHoy; const filas = []; for (let i = 1; i < valores.length; i++) { const fila = valores[i]; if (!fila[1]) continue; // saltar filas vacías // Normalizar la fecha de la celda let fechaCelda = ''; if (fila[0] instanceof Date) { const d = String(fila[0].getDate()).padStart(2, '0'); const m = String(fila[0].getMonth() + 1).padStart(2, '0'); const a = fila[0].getFullYear(); fechaCelda = d + '/' + m + '/' + a; } else { fechaCelda = String(fila[0]).trim(); } if (fechaCelda !== hoyStr) continue; filas.push({ id: parseInt(fila[10]) || (Date.now() + i), nombre: String(fila[1] || ''), ciudad: String(fila[2] || ''), estado: String(fila[3] || 'pendiente'), montoBase: parseFloat(fila[4]) || 0, extras: parseFloat(fila[5]) || 0, flete: parseFloat(fila[6]) || 0, telefono: String(fila[8] || ''), nota: String(fila[9] || '') }); } return ContentService.createTextOutput(JSON.stringify({ ok: true, data: filas, hoy: hoyStr })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.openById(data.sheetId).getActiveSheet(); if (data.action === 'guardar_dia') { const valores = sheet.getDataRange().getValues(); // Borrar filas de hoy (de abajo hacia arriba para no desalinear índices) for (let i = valores.length - 1; i >= 1; i--) { let fechaCelda = ''; if (valores[i][0] instanceof Date) { const d = String(valores[i][0].getDate()).padStart(2, '0'); const m = String(valores[i][0].getMonth() + 1).padStart(2, '0'); const a = valores[i][0].getFullYear(); fechaCelda = d + '/' + m + '/' + a; } else { fechaCelda = String(valores[i][0]).trim(); } if (fechaCelda === data.fecha) sheet.deleteRow(i + 1); } // Escribir envíos data.envios.forEach(env => { sheet.appendRow([ data.fecha, env.nombre, env.ciudad, env.estado, env.montoBase, env.extras || 0, env.flete || 0, env.montoBase + (env.extras || 0), env.telefono || '', env.nota || '', env.id // columna 11: ID para identificar cada envío ]); }); } return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
